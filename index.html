function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // CHANGE "Sheet1" TO YOUR TAB NAME IF DIFFERENT const sheet = ss.getSheetByName("Targets test data") || ss.getSheets()[0]; const data = sheet.getDataRange().getDisplayValues(); const headers = data[0]; const rows = data.slice(1); const jsonData = rows.map(row => { let obj = {}; headers.forEach((header, i) => { obj[header] = row[i] || ""; }); return obj; }); // This specific return format prevents the "Failed to Fetch" error return ContentService.createTextOutput(JSON.stringify(jsonData)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName("Targets test data") || ss.getSheets()[0]; const data = JSON.parse(e.postData.contents); const sheetData = sheet.getDataRange().getValues(); const headers = sheetData[0]; const leadNameIdx = headers.indexOf("Lead Name"); const statusIdx = headers.indexOf("Status"); for (let i = 1; i < sheetData.length; i++) { if (sheetData[i][leadNameIdx] === data.leadName) { sheet.getRange(i + 1, statusIdx + 1).setValue(data.status); // Add other field updates here as needed break; } } return ContentService.createTextOutput(JSON.stringify({result: "success"})) .setMimeType(ContentService.MimeType.JSON); }
